"use client";
import { useState } from 'react';
import CodeEditor from '@/components/CodeEditor';
import ChatSidebar from '@/components/ChatSidebar';

interface Message {
  role: 'user' | 'assistant';
  content: string;
}

export default function Home() {
  // --- STATE ---
  const [question, setQuestion] = useState('');
  const [code, setCode] = useState('');
  const [highlightedLines, setHighlightedLines] = useState<number[]>([]);
  
  const [messages, setMessages] = useState<Message[]>([]);
  const [fullHistory, setFullHistory] = useState<Message[]>([]);

  const [isCorrect, setIsCorrect] = useState(false);
  const [encouragement, setEncouragement] = useState<string>('');
  
  const [isAnalyzing, setIsAnalyzing] = useState(false);
  const [isExecuting, setIsExecuting] = useState(false);

  const [consoleOutput, setConsoleOutput] = useState('');
  const [complexityData, setComplexityData] = useState<any>(null);
  const [language, setLanguage] = useState<'javascript' | 'python' | 'cpp'>('python');

  // --- HANDLERS ---
  const runCode = async () => {
    if (!code.trim()) { alert('Please enter code to run'); return; }
    setIsExecuting(true);
    setConsoleOutput('Compiling & Running...');
    try {
      const response = await fetch('http://localhost:8000/api/execute', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ code, language }),
      });
      if (!response.ok) throw new Error('Execution failed');
      const data = await response.json();
      setConsoleOutput(data.output || 'No output.');
      if(data.is_error) setConsoleOutput((prev) => `Error:\n${data.output}`);
    } catch (error) {
      setConsoleOutput('Error: Connection failed.');
    } finally {
      setIsExecuting(false);
    }
  };

  const analyzeCode = async (historyOverride?: Message[]) => {
    const historyToSend = historyOverride || fullHistory;
    if (!question.trim() || !code.trim()) { alert('Please enter both a question and code'); return; }

    setIsAnalyzing(true);
    setComplexityData(null);
    setConsoleOutput("Analyzing...");

    try {
      const response = await fetch('http://localhost:8000/api/analyze', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          question, code, language: language || "python",
          previously_highlighted: highlightedLines,
          chat_history: historyToSend,
        }),
      });

      if (!response.ok) throw new Error('Failed to analyze code');
      const data = await response.json();

      if (data.execution_output !== undefined) setConsoleOutput(data.execution_output);
      else setConsoleOutput("No output returned.");

      setComplexityData(data.complexity_analysis);
      setHighlightedLines(data.lines_to_highlight);

      const aiMessage: Message = { role: 'assistant', content: data.message };
      setMessages((prev) => [...prev, aiMessage]);
      setFullHistory((prev) => [...prev, aiMessage]);

      setIsCorrect(data.is_correct);
      setEncouragement(data.encouragement || '');
    } catch (error) {
      console.error(error);
      setConsoleOutput('Error: Analysis failed.');
    } finally {
      setIsAnalyzing(false);
    }
  };

  const handleSendMessage = (message: string) => {
    const newUserMsg: Message = { role: 'user', content: message };
    setMessages((prev) => [...prev, newUserMsg]);
    const newHistory = [...fullHistory, newUserMsg];
    setFullHistory(newHistory);
    setTimeout(() => analyzeCode(newHistory), 100);
  };

  const handleClearMessages = () => {
    if (window.confirm('Clear chat UI? (AI context will be remembered)')) {
      setMessages([]);
      setHighlightedLines([]);
      setIsCorrect(false);
      setEncouragement('');
    }
  };

  return (
    <div className="flex flex-col h-screen bg-neutral-950 text-neutral-200 overflow-hidden font-sans selection:bg-blue-500/30">
      
      {/* 1. HEADER */}
      <header className="flex-none px-6 py-3 border-b border-white/5 bg-neutral-900/50 backdrop-blur-md z-10 flex items-center justify-between">
        <div className="flex items-center gap-3">
          <div className="w-9 h-9 bg-gradient-to-br from-blue-600 to-indigo-600 rounded-lg shadow-inner flex items-center justify-center border border-white/10">
            <span className="text-white font-bold text-lg">S</span>
          </div>
          <div>
            <h1 className="text-lg font-bold tracking-tight text-white">SocraCode</h1>
            <p className="text-[11px] text-neutral-400 font-medium tracking-wide uppercase">Socratic Tutor</p>
          </div>
        </div>
        
        {/* Language Selector in Header for cleaner look */}
        <div className="flex items-center gap-3">
            {complexityData && (
                <div className={`flex items-center gap-2 px-3 py-1.5 rounded-full border text-xs font-medium animate-in fade-in zoom-in duration-300 ${
                    complexityData.complexity_score > 10 
                    ? 'bg-red-500/10 border-red-500/20 text-red-400' 
                    : 'bg-emerald-500/10 border-emerald-500/20 text-emerald-400'
                }`}>
                   <span>Complexity Score: {complexityData.complexity_score}</span>
                </div>
            )}
            <select
              value={language}
              onChange={(e) => setLanguage(e.target.value as any)}
              className="bg-neutral-800 border border-white/10 rounded-lg px-3 py-1.5 text-xs text-neutral-300 focus:outline-none focus:ring-2 focus:ring-blue-500/50 transition-all hover:bg-neutral-700"
            >
              <option value="python">Python</option>
              <option value="javascript">JavaScript</option>
              <option value="cpp">C/C++</option>
              <option value="java">Java</option>
            </select>
        </div>
      </header>

      {/* 2. MAIN LAYOUT GRID */}
      <div className="flex-1 p-4 gap-4 flex flex-col min-h-0">
        
        {/* TOP: Question Input (Card) */}
        <div className="flex-none bg-neutral-900/50 border border-white/5 rounded-2xl p-4 flex gap-4 shadow-sm hover:border-white/10 transition-colors">

"use client";
import { useState, useRef, useEffect } from 'react';
import ReactMarkdown from 'react-markdown';
import remarkGfm from 'remark-gfm';
import MermaidDiagram from './MermaidDiagram';

interface Message {
  role: 'user' | 'assistant';
  content: string;
}

interface ChatSidebarProps {
  messages: Message[];
  onSendMessage: (message: string) => void;
  isCorrect: boolean;
  encouragement?: string;
  onClear?: () => void;
}

export default function ChatSidebar({ messages, onSendMessage, isCorrect, encouragement, onClear }: ChatSidebarProps) {
  const [input, setInput] = useState('');
  const messagesEndRef = useRef<HTMLDivElement>(null);

  const scrollToBottom = () => {
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  };

  useEffect(() => {
    scrollToBottom();
  }, [messages]);

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    if (input.trim()) {
      onSendMessage(input);
      setInput('');
    }
  };

  // Custom renderer for Markdown + Mermaid
  const renderContent = (content: string) => {
    const components = {
      code({ node, inline, className, children, ...props }: any) {
        const match = /language-(\w+)/.exec(className || '');
        const isMermaid = match && match[1] === 'mermaid';
        if (!inline && isMermaid) {
          return <MermaidDiagram chart={String(children).replace(/\n$/, '')} />;
        }
        return (
          <code className={`${className} bg-gray-700 px-1 rounded`} {...props}>
            {children}
          </code>
        );
      }
    };
    return (
      <ReactMarkdown remarkPlugins={[remarkGfm]} components={components}>
        {content}
      </ReactMarkdown>
    );
  };

  return (
    <div className="flex flex-col h-full bg-gray-900">
      {/* Header */}
      <div className="flex items-center justify-between px-4 py-3 border-b border-gray-800 bg-gray-900">
        <div className="flex items-center gap-2">
          <div className="w-8 h-8 bg-gradient-to-br from-blue-500 to-purple-600 rounded-full flex items-center justify-center">
            <svg className="w-4 h-4 text-white" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M9.663 17h4.673M12 3v1m6.364 1.636l-.707.707M21 12h-1M4 12H3m3.343-5.657l-.707-.707m2.828 9.9a5 5 0 117.072 0l-.548.547A3.374 3.374 0 0014 18.469V19a2 2 0 11-4 0v-.531c0-.895-.356-1.754-.988-2.386l-.548-.547z" /></svg>
          </div>
          <div>
            <h2 className="text-sm font-semibold text-gray-100">AI Teacher</h2>
            <p className="text-[10px] text-gray-400">Socratic Mode</p>
          </div>
        </div>
        {onClear && (
            <button onClick={onClear} className="text-xs text-red-400 hover:text-red-300">Clear</button>
        )}
      </div>

      {/* Success Banner */}
      {isCorrect && encouragement && (
        <div className="m-4 p-4 bg-green-900/30 border border-green-700/50 rounded-lg">
          <h3 className="text-sm font-bold text-green-400">Success!</h3>
          <p className="text-xs text-green-200 mt-1">{encouragement}</p>
        </div>
      )}

      {/* Messages Area */}
      <div className="flex-1 overflow-y-auto p-4 space-y-4 bg-gray-950">
        {messages.length === 0 && (
          <div className="text-center mt-10 text-gray-600 text-sm">
            <p>Ready to help!</p>
          </div>
        )}
        {messages.map((message, index) => (
          <div key={index} className={`flex ${message.role === 'user' ? 'justify-end' : 'justify-start'}`}>
            <div className={`max-w-[90%] rounded-xl px-4 py-3 shadow-sm text-sm ${
                message.role === 'user' ? 'bg-blue-700 text-white' : 'bg-gray-800 text-gray-200 border border-gray-700'
            }`}>
              <div className="prose prose-invert prose-sm max-w-none">
                {renderContent(message.content)}
              </div>
            </div>
          </div>
        ))}
        <div ref={messagesEndRef} />
      </div>

      {/* Input Area */}
      <div className="p-3 border-t border-gray-800 bg-gray-900">
        <form onSubmit={handleSubmit} className="flex gap-2">
          <input
            type="text"
            value={input}
            onChange={(e) => setInput(e.target.value)}
            placeholder="Explain your logic or ask for a hint..."
            className="flex-1 px-3 py-2 bg-gray-800 border border-gray-700 rounded text-sm text-white focus:outline-none focus:border-blue-500"
          />
          <button type="submit" disabled={!input.trim()} className="px-4 py-2 bg-blue-600 hover:bg-blue-500 text-white rounded text-sm font-medium disabled:opacity-50">
            Send
          </button>
        </form>
      </div>
    </div>
  );
}

"use client";

import { useMemo } from 'react';
import CodeMirror from '@uiw/react-codemirror';
import { javascript } from '@codemirror/lang-javascript';
import { python } from '@codemirror/lang-python';
import { cpp } from '@codemirror/lang-cpp';
import { java } from '@codemirror/lang-java';
import { oneDark } from '@codemirror/theme-one-dark';
import { EditorView, Decoration } from '@codemirror/view';
import { Extension, RangeSetBuilder } from '@codemirror/state';
// 1. Ensure these are imported
import { autocompletion, CompletionContext } from '@codemirror/autocomplete'; 

interface CodeEditorProps {
  value: string;
  onChange: (value: string) => void;
  highlightedLines: number[];
  language?: 'javascript' | 'python' | 'cpp' | 'java';
}

// 2. THE MEGA LIST (Keywords + STL + Types)
function myCppCompletions(context: CompletionContext) {
  let word = context.matchBefore(/\w*/)
  if (!word || (word.from == word.to && !context.explicit)) return null
  
  return {
    from: word.from,
    options: [
      // --- C++ KEYWORDS (Restored) ---
      {label: "int", type: "keyword"},
      {label: "void", type: "keyword"},
      {label: "return", type: "keyword"},
      {label: "if", type: "keyword"},
      {label: "else", type: "keyword"},
      {label: "for", type: "keyword", apply: "for (int i = 0; i < n; i++) {\n\t\n}"},
      {label: "while", type: "keyword"},
      {label: "do", type: "keyword"},
      {label: "switch", type: "keyword"},
      {label: "case", type: "keyword"},
      {label: "break", type: "keyword"},
      {label: "continue", type: "keyword"},
      {label: "const", type: "keyword"},
      {label: "class", type: "keyword"},
      {label: "struct", type: "keyword"},
      {label: "public", type: "keyword"},
      {label: "private", type: "keyword"},
      {label: "include", type: "keyword", apply: "#include < >"},
      {label: "true", type: "keyword"},
      {label: "false", type: "keyword"},
      {label: "auto", type: "keyword"},
      
      // --- STL HEADERS ---
      {label: "iostream", type: "class", apply: "iostream"},
      {label: "vector", type: "class", apply: "vector"},
      {label: "string", type: "class", apply: "string"},
      {label: "algorithm", type: "class", apply: "algorithm"},
      {label: "map", type: "class", apply: "map"},
      {label: "set", type: "class", apply: "set"},
      
      // --- COMMON TYPES & FUNCTIONS ---
      {label: "std", type: "namespace"},
      {label: "cout", type: "variable", detail: "std::cout"},
      {label: "cin", type: "variable", detail: "std::cin"},
      {label: "endl", type: "variable", detail: "std::endl"},
      {label: "vector", type: "type", apply: "vector<int> v;"},
      {label: "string", type: "type"},
      {label: "pair", type: "type"},
      
      // --- SNIPPETS (The "Pro" Stuff) ---
      {label: "main", type: "function", detail: "Main Function", apply: "int main() {\n\t\n\treturn 0;\n}"},
      {label: "us", type: "text", detail: "using namespace std", apply: "using namespace std;"},
    ]
  }
}

export default function CodeEditor({
  value,
  onChange,
  highlightedLines,
  language = 'javascript'
}: CodeEditorProps) {

  const extensions = useMemo(() => {
    // 3. Select Base Language
    const languageExtension = () => {
      if (language === 'python') return python();
      if (language === 'cpp') return cpp();
      if (language === 'java') return java();
      return javascript(); 
    };

    const exts: Extension[] = [
      languageExtension(),
      EditorView.theme({
        '&': { height: '100%', fontSize: '14px' },
        '.cm-scroller': { overflow: 'auto', fontFamily: 'monospace' },
        '.cm-content': { padding: '12px 0' },
        '.cm-line': { padding: '0 8px' },
        '.highlighted-line': { 
            backgroundColor: 'rgba(135, 206, 235, 0.15)', 
            borderLeft: '3px solid #60a5fa',
            paddingLeft: '5px !important'
        }
      }),
    ];

    // 4. Force Autocomplete to Active
    // If C++, use our custom list. 
    // If others, use default autocompletion() which enables standard keywords.
    if (language === 'cpp') {
       exts.push(autocompletion({ override: [myCppCompletions] }));
    } else {
       exts.push(autocompletion());
    }

    // Line Highlighting Logic
    if (highlightedLines.length > 0) {
      exts.push(
        EditorView.decorations.of((view: EditorView) => {
          const builder = new RangeSetBuilder<Decoration>();
          const lineDeco = Decoration.line({ class: 'highlighted-line' });
          for (const ln of highlightedLines) {
            if (ln >= 1 && ln <= view.state.doc.lines) {
              const line = view.state.doc.line(ln);
              builder.add(line.from, line.from, lineDeco);
            }
          }
          return builder.finish();
        })
      );
    }

    return exts;
  }, [language, highlightedLines]);

  return (
    <div className="h-full w-full rounded-lg overflow-hidden border border-white/5 bg-[#1e1e1e]">
      <CodeMirror
        value={value}
        height="100%"
        theme={oneDark}
        extensions={extensions}
        onChange={(val) => onChange(val)}
        // We keep basicSetup for other niceties (history, brackets), 
        // but we handled autocompletion explicitly above.
        basicSetup={{
          lineNumbers: true,
          highlightActiveLineGutter: true,
          closeBrackets: true,
          history: true,
          indentOnInput: true,
          autocompletion: false,
        }}
        style={{ height: '100%', fontSize: '14px' }}
      />
    </div>
  );
}

import React, { useEffect, useRef, useState } from 'react';
import mermaid from 'mermaid';

mermaid.initialize({
  startOnLoad: false,
  theme: 'base',
  securityLevel: 'loose',
  suppressErrorRendering: true,
  themeVariables: {
    darkMode: true,
    primaryColor: '#1e293b',
    lineColor: '#38bdf8',
    textColor: '#f8fafc',
    mainBkg: '#1e293b',
    nodeBorder: '#94a3b8',
  }
});


// ... (Keep your mermaid.initialize code here) ...

export default function MermaidDiagram({ chart }: { chart: string }) {
  const ref = useRef<HTMLDivElement>(null);
  const [isError, setIsError] = useState(false);
  const [svgCode, setSvgCode] = useState<string>('');

  useEffect(() => {
    if (!chart || !ref.current) return;

    const render = async () => {
      try {
        setIsError(false);
        const id = `mermaid-${Math.random().toString(36).substr(2, 9)}`;
        await mermaid.parse(chart);
        const { svg } = await mermaid.render(id, chart);
        setSvgCode(svg);
        if (ref.current) ref.current.innerHTML = svg;
      } catch (error) {
        console.error("Mermaid Syntax Error:", error);
        setIsError(true);
      }
    };
    render();
  }, [chart]);

  const handleOpenNewTab = () => {
    if (!svgCode) return;
    const htmlContent = `
      <!DOCTYPE html>
      <html>
        <head>
          <title>Mermaid Diagram</title>
          <style>
            body { background-color: #0f172a; margin: 0; display: flex; justify-content: center; align-items: center; min-height: 100vh; padding: 20px; }
            svg { max-width: 100%; height: auto; }
            /* Force colors */
            .node rect, .node circle { fill: #1e293b; stroke: #94a3b8; }
            .node .label { color: #f8fafc; }
            text { fill: #f8fafc !important; font-family: sans-serif; }
            path, line { stroke: #38bdf8 !important; stroke-width: 2px; }
          </style>
        </head>
        <body>${svgCode}</body>
      </html>
    `;
    const blob = new Blob([htmlContent], { type: 'text/html' });
    const url = URL.createObjectURL(blob);
    window.open(url, '_blank');
  };

  if (isError) return null;

  return (
    <div className="my-4 border border-gray-700 rounded-lg overflow-hidden bg-gray-800">
      
      {/* 1. VISIBLE HEADER BAR */}
      <div className="flex justify-between items-center px-3 py-2 bg-gray-900 border-b border-gray-700">
        <span className="text-xs text-gray-400 font-mono">Visualization</span>
        <button 
          onClick={handleOpenNewTab}
          className="flex items-center gap-1 text-xs text-blue-400 hover:text-blue-300 transition-colors"
        >
          <svg className="w-3 h-3" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M10 6H6a2 2 0 00-2 2v10a2 2 0 002 2h10a2 2 0 002-2v-4M14 4h6m0 0v6m0-6L10 14"></path></svg>
          Open in New Tab
        </button>
      </div>
